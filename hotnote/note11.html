centos设置vi编辑高亮显示

 







 CentOS里只默认安装了vi，没有vim,所以syntax高亮不起作用。
 #下载vim
 [root@zkyml ~]# yum -y install vim
 #编辑bash配置设置vim的别名为vi
 [root@zkyml ~]# vi /etc/bashrc
 alias vi='vim' 
#编辑vim的配置,显示行号 
[root@zkyml ~]# vi /etc/vimrc 
set nu 
syntax on #如果有就不需要了 
#重新加载生效 
[root@zkyml ~]# source /etc/bashrc
