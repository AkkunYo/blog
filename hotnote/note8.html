centos7新装后安装脚本

	

 
 

#!/system/bin/sh
yum install -y vim net-tools wget
yum -y install httpd httpd-manual mod_ssl mod_perl mod_auth_mysql
chkconfig httpd on
service httpd start

yum install -y mysql mysql-server
service mysqld start
chkconfig mysqld on
yum -y install php php-mysql php-fpm
yum -y install gd php-gd gd-devel php-xml php-common php-mbstring php-ldap php-pear php-xmlrpc php-imap
service httpd restart

systemctl stop firewalld.service #停止firewall
systemctl disable firewalld.service #禁止firewall开机启动
yum -y remove firewalld
#firewall-cmd --state #查看默认防火墙状态（关闭后显示notrunning，开启后显示running）
yum -y install iptables iptables-service
systemctl restart iptables.service #最后重启防火墙使配置生效
systemctl enable iptables.service #设置防火墙开机启动

#添加mono环境下载源,安装nat123
yum install -y yum-utils
rpm --import "http://keyserver.ubuntu.com/pks/lookup?op=get&search=0x3FA7E0328081BFF6A14DA29AA6A19B38D3D831EF"
yum-config-manager --add-repo http://download.mono-project.com/repo/centos/
yum makecache
yum install -y mono-complete #或者(mono-devel>>monodevelop>>monoDevelop)
yum update -y

cd  /mnt
wget http://www.nat123.com/down/nat123linux.tar.gz
tar -zxvf nat123linux.tar.gz
mono nat123linux.sh
